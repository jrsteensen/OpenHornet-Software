157void onApuControlSwChange(unsigned int newValue) {
158 apuSwitchOn = newValue;
159 if (apuSwitchOn == 0 && bleedAirAugState == 1) { // If APU switch turned off and Bleed Air Augmentation up, then pull solenoid.
160 bleedAirAugState = 0;
161 digitalWrite(BA_SOLENOID, HIGH); // pull down.
162 delay(200); //wait a moment
163 digitalWrite(BA_SOLENOID, LOW); // turn off solenoid.
164 }
165}
166DcsBios::IntegerBuffer apuControlSwBuffer(FA_18C_hornet_APU_CONTROL_SW, onApuControlSwChange);

168void onBleedAirPullChange(unsigned int newValue) {
169 if (bleedAirAugState != newValue) {
170 bleedAirAugState = newValue;
171 switch (newValue) {
172 case 0: // If BA pushed virtually within sim, pull down in pit
173 digitalWrite(BA_SOLENOID, HIGH); // pull down.
174 delay(200); //wait a moment
175 digitalWrite(BA_SOLENOID, LOW); // turn off solenoid.
176 break;
177 case 1:
179 if (apuSwitchOn == 0 || (wowLeft == wowRight == wowNose == 0)) { //then pull down, ELSE do nothing.
180 digitalWrite(BA_SOLENOID, HIGH); // pull down.
181 delay(200); //wait a moment
182 digitalWrite(BA_SOLENOID, LOW); // turn off solenoid.
183 }
184 break;
185 }
186 }
187}
188DcsBios::IntegerBuffer bleedAirPullBuffer(FA_18C_hornet_BLEED_AIR_PULL, onBleedAirPullChange);

242void loop() {
243 DcsBios::loop();
244
246 bleedAirKnob.pollThisInput();
247
248 if (bleedAirAugState == 1) { // If bleed air aug is on...
249 if ((rpmL >= 95 && rpmR >= 95) || (leftGenState == 0 && rightGenState == 0)) { // If over 95% rpm or both generators off pull down on bleed air aug
250 digitalWrite(BA_SOLENOID, HIGH); // pull down.
251 delay(200); //wait a moment
252 digitalWrite(BA_SOLENOID, LOW); // turn off solenoid.
253 bleedAirAugState = 0;
254 }
255 }
