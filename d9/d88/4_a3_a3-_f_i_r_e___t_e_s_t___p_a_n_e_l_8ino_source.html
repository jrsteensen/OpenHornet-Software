211void loop() {
212
213 //Run DCS Bios loop function
214 DcsBios::loop();
215
216 // Brake Handle Logic:
217 bool eBrkSwitchState = !digitalRead(BRKH_EMERG); // read the emergency brake switch state
218 bool pBrkSwitchState = !digitalRead(BRKH_PARK); // read the parking brake switch state
219
220 switch (brakeState) { // brakeState = 1:Park, 2:Emergency, or 0:Off
221 case 0: // brakeState == 0, brakes are off determine if pulled and which direction
222 if (eBrkSwitchState == 1) { // Emerg brake switch state == 1, Brake handle pulled for emergency brake
223 sendDcsBiosMessage("EMERGENCY_PARKING_BRAKE_PULL", "0"); // DcsBios message to pull handle
224 brakeState = 2; // Update brake state to emergency brakes applied
225 }
226 if (pBrkSwitchState == 1) { // Parking brake switch state == 1, brake handle pulled for parking brake
227 sendDcsBiosMessage("EMERGENCY_PARKING_BRAKE_ROTATE", "1"); // Rotate parking brake handle to parking / vertical position
228 delay(50);
229 sendDcsBiosMessage("EMERGENCY_PARKING_BRAKE_PULL", "0"); // DcsBios message to pull handle
230 brakeState = 1; // Update brake state to parking brakes applied
231 }
232 break;
233 case 1: // brakeState == 1, Parking brake on
234 if (pBrkSwitchState == 0) { // Parking brake switch state == 0, brake handle pushed in
235 sendDcsBiosMessage("EMERGENCY_PARKING_BRAKE_ROTATE", "2"); // DcsBios message to rotate the brake handle to the 'release' position (not modeled in the physical pit)
236 delay(50); // delay for the animation to complete
237 sendDcsBiosMessage("EMERGENCY_PARKING_BRAKE_PULL", "1"); // DcsBios message to push the brake handle in
238 delay(50); // delay for the animation to complete
239 sendDcsBiosMessage("EMERGENCY_PARKING_BRAKE_ROTATE", "0"); // DcsBios message to rotate the brake handle to the emergency / horizontal position
240 brakeState = 0; // Update brake state to off
241 }
242 break;
243 case 2: // brakeState == 2, Emergency brake on
244 if (eBrkSwitchState == 0) { // Emergency brake swtich state == 0, brake handle pushed in
245 sendDcsBiosMessage("EMERGENCY_PARKING_BRAKE_PULL", "1"); // DcsBios message to push the brake handle in
246 brakeState = 0; // Update brake state to off
247 }
248 break;
249 }
250
251 // Brake Pressue Gauge Logic:
252 if (abs(brakePressureValue - brakePressurePos) > 2) { // difference is greater than 2 steps.
253 if ((brakePressureValue - brakePressurePos) > 0) { // If value is greater than current position rotate needle clockwise
254 stepperBrakePressure.step(-1); // Step one value clockwise
255 brakePressurePos++; // update brake pressure position
256 }
257 if ((brakePressureValue - brakePressurePos) < 0) { // If value is less than current position rotate needle counter-clockwise
258 stepperBrakePressure.step(1); // Step one value counter-clockwise
259 brakePressurePos--; // update brake pressure position
260 }
261 }
262}
