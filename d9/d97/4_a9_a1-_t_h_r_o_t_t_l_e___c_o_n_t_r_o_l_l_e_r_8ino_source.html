213void setup() {
214
215 // Run DCS Bios setup function
216 DcsBios::setup();
217
218 Wire.begin(SDA, SCL);
219 Serial.begin(115200);
220
221 outboardThrottle.init();
222 inboardThrottle.init();
223 inboardThrottle.setRotationDirection(-1); // reverse hall sensor rotation so both throttles read as increasing when moving forward
224
225#ifdef SET_THROTTLE_ZERO
226 //throttle needs to be all the way back to ground idle when plugged into the computer to set the zero value.
227 outboardThrottle.setZeroFromCurrentPosition();
228 inboardThrottle.setZeroFromCurrentPosition();
229#endif
230
231 pinMode(OUTBD_MAX_LIMIT, INPUT_PULLUP);
232 pinMode(OUTBD_IDLE_LIMIT, INPUT_PULLUP);
233 pinMode(INBD_MAX_LIMIT, INPUT_PULLUP);
234 pinMode(INBD_IDLE_LIMIT, INPUT_PULLUP);
235 pinMode(EXT_LTS, INPUT_PULLUP);
236 pinMode(RAID_FLIR, INPUT_PULLUP);
237 pinMode(ATC_ENGAGE, INPUT_PULLUP);
238
239 pinMode(IDLE_SOL_SIG, OUTPUT);
240 pinMode(MAX_SOL_SIG, OUTPUT);
241
242 digitalWrite(IDLE_SOL_SIG, LOW);
243 digitalWrite(MAX_SOL_SIG, LOW);
244
245 Joystick.setXAxisRange(0, 1024); // TDC X-axis
246 Joystick.setYAxisRange(0, 1024); // TDC Y-axis
247 Joystick.setZAxisRange(0, 1024); // Radar Elevation
248 Joystick.setRxAxisRange(0, 65535); // Outboard Throttle Arm
249 Joystick.setRyAxisRange(0, 65535); // Inboard Throttle Arm
250 Joystick.begin();
251}

278void loop() {
279
280 //Run DCS Bios loop function
281 DcsBios::loop();
282
283 uint32_t temp; // temp value to hold the analog reads in preparation of doing logic.
284
285 outboardThrottle.update(); // update the outboard hall sensor to prep for read.
286 temp = outboardThrottle.readRawAngle21(); // read outboard hall sensor
287 Joystick.setRxAxis(mapHallSensor(temp, 0, 838440, 0, 65535)); //0 and 838440 came from reading the Serial Monitor for the min/max values to then plug into this line.
288 // Uncomment the code below if you wish to pass the outboard throttle's raw values to the serial monitor
289 //Serial.print("outbThrottle: ");
290 //Serial.print(temp);
291 //Serial.print("\n");
292
293 inboardThrottle.update(); // update the inboard hall sensor to prep for read
294 temp = inboardThrottle.readRawAngle21(); // read inboard hall sensor
295 Joystick.setRyAxis(mapHallSensor(temp, 0, 838440, 0, 65535)); //0 and 838440 came from reading the Serial Monitor for the min/max values to then plug into this line.
296 // Uncomment the code below if you wish to pass the inboard throttle's raw values to the serial monitor
297 //Serial.print(" inbThrottle: ");
298 //Serial.print(temp);
299 //Serial.print("\n");
300
301 // determine if max limit switches are included or not
302 #ifndef DISABLE_MAX_LIMIT_SWITCHES
303 Joystick.setButton(0, !digitalRead(OUTBD_MAX_LIMIT));
304 Joystick.setButton(1, !digitalRead(INBD_MAX_LIMIT));
305 #endif
306
307 Joystick.setButton(2, !digitalRead(OUTBD_IDLE_LIMIT));
308 Joystick.setButton(3, !digitalRead(INBD_IDLE_LIMIT));
309 Joystick.setButton(4, !digitalRead(EXT_LTS));
310 Joystick.setButton(5, !digitalRead(RAID_FLIR));
311 Joystick.setButton(6, !digitalRead(ATC_ENGAGE));
312
313 int index = 0; // Initialize the index to keep track of which inner grip button we're reading
314 Wire.requestFrom(49, 17); // request inner grip pro-mini return current button state info
315
316
317
318 while (Wire.available()) { // While something to read from the inner grip
319 if (index < JOY_X) { // The first sets of values are button presses until we get to axis values
320 Joystick.setButton(index + 7, Wire.read()); // inner throttle's digital button reads, set as joystick button state
321 }
322 // Remaining Index values are inner grip analog reads
323 else if (index == JOY_X) { // TDC X-axis
324 byte tempLow = Wire.read(); // read low byte
325 byte tempHi = Wire.read(); // read high byte
326 temp = (tempHi << 8) + tempLow; //rebuild the TDC X-axis value
327 Joystick.setXAxis(map(temp, 0, 65000, 0, 1024));
328
329 } else if (index == JOY_Y) { // TDC Y-axis
330 byte tempLow = Wire.read(); // read low byte
331 byte tempHi = Wire.read(); // read high byte
332 temp = (tempHi << 8) + tempLow; //rebuild the TDC Y-axis value
333 Joystick.setYAxis(map(temp, 0, 65000, 0, 1024));
334
335 } else if (index == ANTENNA_ELEVATION) {
336 byte tempLow = Wire.read(); // read low byte
337 byte tempHi = Wire.read(); // read high byte
338 temp = (tempHi << 8) + tempLow; //rebuild the Antenna axis value
339 Joystick.setZAxis(map(temp, 754, 978, 0, 1024)); // mapped values determined by reading the retured elevation on the serial monitor
340 // enable the lines below to see the Elevation knob's analog read values.
341 //Serial.print(" Elevation: ");
342 //Serial.print((tempHi << 8) + tempLow);
343 //Serial.print("\n");
344 }
345 index++; // increment the index for the next iteration through the while loop
346 }
347
348#ifndef DISABLE_IDLE_SOL_SIG
349 if (wowLeft == wowRight == wowNose == true) {
350 digitalWrite(IDLE_SOL_SIG, HIGH);
351 } else {
352 digitalWrite(IDLE_SOL_SIG, LOW);
353 }
354#endif
355
356#ifndef DISABLE_MAX_SOL_SIG
357 if ((wowLeft == wowRight == wowNose == true) && ((launchBarExtended == true) || (arrestingHookUp == false))) {
358 digitalWrite(MAX_SOL_SIG, LOW); // With weight-on-wheels, if the lanuch bar is extended or the hook is down then raise afterburner lockout / detent to prevent inadvertent afterburner selection.
359 } else {
360 digitalWrite(MAX_SOL_SIG, HIGH); // Otherwise pull the afterburner lockout / detent.
361 }
362#endif
363}

371long mapHallSensor(unsigned long long x, unsigned long long in_min, unsigned long long in_max, unsigned long long out_min, unsigned long long out_max) {
372 const unsigned long long run = in_max - in_min;
373 if(run == 0){
374 log_e("map(): Invalid input range, min == max");
375 return -1; // AVR returns -1, SAM returns 0
376 }
377 const unsigned long long rise = out_max - out_min;
378 const unsigned long long delta = x - in_min;
379 return (delta * rise) / run + out_min;
380}
