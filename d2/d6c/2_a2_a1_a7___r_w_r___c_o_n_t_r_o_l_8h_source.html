82 RwrControlPanel(int startIndex, CRGB* ledStrip) {
83 panelStartIndex = startIndex;

96 // State variables
97 bool rwrPanelOn; // RWR panel power state
98 bool rwrFailActive; // RWR fail indicator state
99 bool rwrOffsetEnableActive; // RWR offset enable state
100 bool rwrSpecialEnableActive; // RWR special enable state
101 bool rwrLimitDisplayActive; // RWR limit display state
106 void updateLedColors() {
107 if (!rwrPanelOn) {
108 // When RWR panel is OFF, all indicator LEDs are black
109 setIndicatorColor(LED_RWR_POWER, NVIS_BLACK);
110 setIndicatorColor(LED_RWR_BIT, NVIS_BLACK);
111 setIndicatorColor(LED_RWR_BIT_FAIL, NVIS_BLACK);
112 setIndicatorColor(LED_RWR_OFFSET, NVIS_BLACK);
113 setIndicatorColor(LED_RWR_OFFSET_EN, NVIS_BLACK);
114 setIndicatorColor(LED_RWR_DISPLAY, NVIS_BLACK);
115 setIndicatorColor(LED_RWR_LIMIT, NVIS_BLACK);
116 setIndicatorColor(LED_RWR_SPECIAL, NVIS_BLACK);
117 setIndicatorColor(LED_RWR_SPECIAL_EN, NVIS_BLACK);
118 } else {
119 // When RWR panel is ON, set colors based on state
120 // RWR POWER
121 setIndicatorColor(LED_RWR_POWER, NVIS_GREEN_A);
122 // RWR BIT and RWR BIT FAIL
123 if (rwrFailActive) {
124 setIndicatorColor(LED_RWR_BIT, NVIS_RED);
125 setIndicatorColor(LED_RWR_BIT_FAIL, NVIS_RED);
126 } else {
127 setIndicatorColor(LED_RWR_BIT, NVIS_WHITE);
128 setIndicatorColor(LED_RWR_BIT_FAIL, NVIS_WHITE);
129 }
130
131 // RWR OFFSET and RWR OFFSET EN
132 if (rwrOffsetEnableActive) {
133 setIndicatorColor(LED_RWR_OFFSET, NVIS_GREEN_A);
134 setIndicatorColor(LED_RWR_OFFSET_EN, NVIS_GREEN_A);
135 } else {
136 setIndicatorColor(LED_RWR_OFFSET, NVIS_WHITE);
137 setIndicatorColor(LED_RWR_OFFSET_EN, NVIS_WHITE);
138 }
139
140 // RWR SPECIAL and RWR SPECIAL EN
141 if (rwrSpecialEnableActive) {
142 setIndicatorColor(LED_RWR_SPECIAL, NVIS_GREEN_A);
143 setIndicatorColor(LED_RWR_SPECIAL_EN, NVIS_GREEN_A);
144 } else {
145 setIndicatorColor(LED_RWR_SPECIAL, NVIS_WHITE);
146 setIndicatorColor(LED_RWR_SPECIAL_EN, NVIS_WHITE);
147 }
148
149 // RWR LIMIT DISPLAY
150 if (rwrLimitDisplayActive) {
151 setIndicatorColor(LED_RWR_DISPLAY, NVIS_GREEN_A);
152 setIndicatorColor(LED_RWR_LIMIT, NVIS_GREEN_A);
153 } else {
154 setIndicatorColor(LED_RWR_DISPLAY, NVIS_WHITE);
155 setIndicatorColor(LED_RWR_LIMIT, NVIS_WHITE);
156 }
157 }
158 }
