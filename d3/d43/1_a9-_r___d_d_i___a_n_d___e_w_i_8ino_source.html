163void setup() {
164
165 // Run DCS Bios setup function
166 DcsBios::setup();
167
168 pinMode(DDI_BACK_LIGHT, OUTPUT); // set backlighting pin mode to output
169 analogWrite(DDI_BACK_LIGHT, 0); // turn off backlighting
170
171 pinMode(DDI_IRQ, OUTPUT); // Set IRQ pinmode to match sample code to avoid controls becomming unresponsive issue
172
177 for (int i = 0; i < sizeof(lastBtnState) / sizeof(lastBtnState[0]); i++) {
178 lastBtnState[i] = 0;
179 }
180
185 for (int i = 0; i < sizeof(ddiButtons) / sizeof(ddiButtons[0]); i++) {
186 ddiButtons[i].Begin();
187 for (int j = 0; j < 5; j++) {
188 ddiButtons[i].PinMode(j, INPUT);
189 }
190 }
191}

201void loop() {
202
203 //Run DCS Bios loop function
204 DcsBios::loop();
205
210 for (int i = 0; i < sizeof(ddiButtons) / sizeof(ddiButtons[0]); i++) { // RIGHT = 0, Top = 1, Right = 2, Bottom = 3
211 inputRegister[i] = ddiButtons[i].ReadAll();
212
217 for (int j = 0; j < 5; j++) {
218
219 if (i == 1 || i == 2) { //button order reversed, adjust the index accordingly.
220 index = ((4 - j) + 5 * i);
221 } else {
222 index = (j + 5 * i);
223 }
224
230 bool btnState = (inputRegister[i] >> (4 - j)) & 1;
231
232 if (btnState != lastBtnState[index]) {
233 lastDebounceTime[index] = millis();
234 }
235
236 if ((millis() - lastDebounceTime[index]) > debounceDelay) {
237 if (btnState != buttonState[index]) {
238 buttonState[index] = btnState;
239 char btnName[14];
240 sprintf(btnName, "RIGHT_DDI_PB_%02d", index + 1);
241 DcsBios::tryToSendDcsBiosMessage(btnName, btnState == 1 ? "0" : "1");
242 }
243 }
244 lastBtnState[index] = btnState;
245 }
246 }
