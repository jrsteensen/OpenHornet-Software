145void requestEvent() {
146
147 Wire.write(!digitalRead(CAGE_UNCAGE));
148 Wire.write(!digitalRead(SPEEDBRAKE_RETRACT));
149 Wire.write(!digitalRead(SPEEDBRAKE_EXTEND));
150 Wire.write(!digitalRead(COUNTERMEASURES_AFT));
151 Wire.write(!digitalRead(COUNTERMEASURES_FWD));
152 Wire.write(!digitalRead(D));
153 Wire.write(!digitalRead(C));
154 Wire.write(!digitalRead(B));
155 Wire.write(!digitalRead(A));
156 Wire.write(!digitalRead(PUSH));
157 Wire.write(!digitalRead(JOY_SW));
158
159 int temp = analogRead(JOY_X); // get the analog read of the TDC X azis.
160 Wire.write(lowByte(temp)); // write out the low byte to the controller
161 Wire.write(highByte(temp)); // write out the high byte to the controller
162
163 temp = analogRead(JOY_Y); // get the analog read of the TDC Y azis.
164 Wire.write(lowByte(temp)); // write out the low byte to the controller
165 Wire.write(highByte(temp)); // write out the high byte to the controller

168 Wire.write(lowByte(temp)); // write out the low byte to the controller
169 Wire.write(highByte(temp)); // write out the high byte to the controller
