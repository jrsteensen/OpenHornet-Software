460void setup() {
461 #if (not defined( DCSBIOS_RS485_SLAVE) || not defined(DCSBIOS_RS485_MASTER )) // Disable RS485 TRANSCEIVER IF NOT BEING USED
462 pinMode(TXENABLE_PIN, OUTPUT); // WILL DISABLE TX ONLY
463 digitalWrite(TXENABLE_PIN, HIGH); // SET TX HIGH TO DISABLE TX ON TRANSCEIVER
464 #endif
465
466 //OLEDS ALTIMETER SETUP
467 Wire.begin();
468 for (uint8_t t = 0; t < 8; t++) {
469 tcaselect(t);
470
471 for (uint8_t addr = 0; addr <= 127; addr++) {
472 //if (addr == TCAADDR) continue;
473
474 uint8_t data;
475 if (!twi_writeTo(addr, &data, 0, 1, 1)) {
476 // Had to comment out these debugging messages as they created a conflict with the IRQ definition in DCS BIOS
477 }
478 }
479 }
480
481 tcaselect(BARO_OLED_Port);
482 u8g2_BARO.begin();
483 u8g2_BARO.clearBuffer();
484 u8g2_BARO.setFont(u8g2_font_fub14_tr);
485 u8g2_BARO.sendBuffer();
486
487 tcaselect(ALT_OLED_Port);
488 u8g2_ALT.begin();
489 u8g2_ALT.clearBuffer();
490 u8g2_ALT.setFont(u8g2_font_fub20_tr);
491 u8g2_ALT.sendBuffer();
492
493 updateALT("0", "0");
494 updateBARO("2992");
495
496 //AIRSPEED STEPPER HOMING
497 pinMode(AIR_ZERO_SENSE_PIN, INPUT_PULLUP);
498 stepperSTANDBY_AIR.setSpeed(60);
499 stepperSTANDBY_AIR.step(1000);
500 for (int i = 0; i <= 2000; i++) {
501 delay(1);
502 stepperSTANDBY_AIR.step(1);
503 if (digitalRead(AIR_ZERO_SENSE_PIN) == 0) {
504 delay(500);
505 stepperSTANDBY_AIR.step(AIR_OFFSET_TO_ZERO_POINT);
506 break;
507 }
508 }
509
510 posAIR - 0;
511 AIR = map(0, 0, 65535, 0, 720);
512
513 //ALTIMETER STEPPER HOMING
514 pinMode(ALT_ZERO_SENSE_PIN, INPUT_PULLUP);
515 stepperSTANDBY_ALT.setSpeed(60);
516 stepperSTANDBY_ALT.step(1000);
517 for (int i = 0; i <= 2000; i++) {
518 delay(1);
519 stepperSTANDBY_ALT.step(1);
520 if (digitalRead(ALT_ZERO_SENSE_PIN) == 0) {
521 delay(500);
522 stepperSTANDBY_ALT.step(ALT_OFFSET_TO_ZERO_POINT);
523 break;
524 }
525 }
526
527 posALT = 0;
528 ALT = 0;
529
530 //VVI STEPPER HOMING
531 pinMode(VVI_ZERO_SENSE_PIN, INPUT_PULLUP);
532 stepperSTANDBY_VVI.setSpeed(60);
533 stepperSTANDBY_VVI.step(1000);
534 for (int i = 0; i <= 2000; i++) {
535 delay(1);
536 stepperSTANDBY_VVI.step(1);
537 if (digitalRead(VVI_ZERO_SENSE_PIN) == 0) {
538 delay(500);
539 stepperSTANDBY_VVI.step(VVI_OFFSET_TO_ZERO_POINT);
540 break;
541 }
542 }
543
544 posVVI - 0;
545 VVI = map(0, 0, 65535, 0, 720);
546
547 FastLED.addLeds<WS2812B, BACKLIGHT_PIN, RGB>(ws2812, BACKLIGHT_COUNT); // GRB ordering is typical
548 fill_solid(ws2812, BACKLIGHT_COUNT, CRGB::Black);
549 FastLED.show();
550 fill_solid(ws2812, BACKLIGHT_COUNT, CRGB::LightGreen);
551
552 DcsBios::setup();
553}

559void loop() {
560 DcsBios::loop();
561
562 //UPDATE THE OLEDS EVERY 100 MS TO AVOID LAG WITH STEPPERS
563 static unsigned long lastOLEDUpdate = 0;
564 unsigned long currentMillis = millis();
565
566 if (currentMillis - lastOLEDUpdate >= 100) { //EVERY 100 MS
567 lastOLEDUpdate = currentMillis;
568
569 if (BaroUpdated == true) buildBAROString();
570 if (AltUpdated == true) updateALT(Alt10000s, Alt1000s);
571 }
572
573 //MOVE AIRSPEED STEPPER
574 valAIR = AIR;
575 if (abs(valAIR - posAIR) > 2) { //IF DIFFERENCE IS GREATER THAN 2 STEPS
576 if ((valAIR - posAIR) > 0) {
577 stepperSTANDBY_AIR.step(1); //MOVE CLOCKWISE
578 posAIR++;
579 }
580 if ((valAIR - posAIR) < 0) {
581 stepperSTANDBY_AIR.step(-1); //MOVE COUNTERCLOCKWISE
582 posAIR--;
583 }
584 }
585
586 //MOVE ALTIMETER STEPPER
587 valALT = ALT;
588 DiffvalposALT = valALT - posALT;
589
590 if (DiffvalposALT > 359) {
591 DiffvalposALT -= 720; //ALLOW LOOPBACK
592 }
593 if (DiffvalposALT < -359) {
594 DiffvalposALT += 720; //ALLOW LOOPBACK
595 }
596
597 if (DiffvalposALT != 0) {
598 if (DiffvalposALT > 0) {
599 stepperSTANDBY_ALT.step(1); //MOVE CLOCKWISE
600 posALT++;
601 if (posALT >= 720) posALT = 0; //LOOPBACK TO KEEP posALT BETWEEN 0 AND 720 WHEN THE NEEDLE IS PASSING 0 FEET
602 } else {
603 stepperSTANDBY_ALT.step(-1); //MOVE COUNTERCLOCKWISE
604 posALT--;
605 if (posALT < 0) posALT = 719; //LOOPBACK TO KEEP posALT BETWEEN 0 AND 720 WHEN THE NEEDLE IS PASSING 0 FEET
606 }
607 }
608
609 //AUTO RECALIBRATION IF WE PASS THROUGH PHYSICAL ZERO EVERY 2 MINUTES TO AVOID PERMANENT OFFSET
610 unsigned long now = millis();
611 if (now - LastALTAutoRecalTime >= 120000UL) { //2 MINUTES = 120000 MS
612 GAUGE_RECAL_READY = true; //READY TO RECALIBRATE AT NEXT ZERO
613 }
614
615 bool ALT_ZERO_CURRENT_STATE = digitalRead(ALT_ZERO_SENSE_PIN);
616
617 if (GAUGE_RECAL_READY && ALT_ZERO_LAST_STATE == HIGH && ALT_ZERO_CURRENT_STATE == LOW) {
618 posALT = 0;
619 LastALTAutoRecalTime = now;
620 GAUGE_RECAL_READY = false; //WAIT NEXT TWO MINUTES BEFORE ANOTHER RECAL
621 }
622
623 ALT_ZERO_LAST_STATE = ALT_ZERO_CURRENT_STATE;
624
625 //MOVE VVI STEPPER
626 valVVI = VVI;
627 if (abs(valVVI - posVVI) > 2) { //IF DIFFERENCE IS GREATER THAN 2 STEPS
628 if ((valVVI - posVVI) > 0) {
629 stepperSTANDBY_VVI.step(1); //MOVE CLOCKWISE
630 posVVI++;
631 }
632 if ((valVVI - posVVI) < 0) {
633 stepperSTANDBY_VVI.step(-1); //MOVE COUNTERCLOCKWISE
634 posVVI--;
635 }
636 }
637}
