62enum class DcsState {
63 EXITED, // DCS not running, in menu, or crashed
64 PAUSED, // Mission loaded but paused (no heartbeat for 10s)
65 GROUND_COLD, // On ground, at least one engine below 50% RPM
66 GROUND_HOT, // On ground, both engines above 50% RPM
67 AIRBORNE // In flight (weight off wheels)
68};

70// ─────────────────────────────────────────────────────────────────────────────
71// State tracking variables
72// ─────────────────────────────────────────────────────────────────────────────
73static char acftName[25] = ""; // Aircraft name (24 chars + null terminator)
74static int rpmLeft = 0; // Left engine RPM percentage
75static int rpmRight = 0; // Right engine RPM percentage
76static unsigned int wowLeft = 1; // Weight on wheels left (1=ground, 0=air)
77static unsigned int wowRight = 1; // Weight on wheels right (1=ground, 0=air)
78
79// Heartbeat tracking
80static unsigned int currDcsHeartbeat = 0;
81static unsigned int prevDcsHeartbeat = 0;
82static unsigned long dcsLastUpdateTime = 0;
83
84// Timing constants
85static const unsigned long PAUSED_TIMEOUT_MS = 10000; // 10 seconds -> PAUSED
86static const unsigned long EXITED_TIMEOUT_MS = 1800000; // 30 minutes -> EXITED (crash detection)
87static const int RPM_THRESHOLD = 50; // RPM threshold for hot/cold
88
89// ─────────────────────────────────────────────────────────────────────────────
90// State determination function
91// ─────────────────────────────────────────────────────────────────────────────
92
100DcsState getDcsState() {
101 unsigned long currentTime = millis();
102 unsigned long timeSinceUpdate = currentTime - dcsLastUpdateTime;
103
104 // Update heartbeat tracking if changed
105 if (currDcsHeartbeat != prevDcsHeartbeat) {
106 dcsLastUpdateTime = currentTime;
107 prevDcsHeartbeat = currDcsHeartbeat;
108 timeSinceUpdate = 0;
109 }
110
111 // Check if aircraft name is empty (DCS in menu or not running)
112 bool hasAircraft = (acftName[0] != '\0');
113
114 // ── EXITED: No aircraft OR timeout exceeded (crash detection) ──
115 if (!hasAircraft || timeSinceUpdate >= EXITED_TIMEOUT_MS) {
116 return DcsState::EXITED;
117 }
118
119 // ── PAUSED: Aircraft loaded but heartbeat stalled ──
120 if (timeSinceUpdate >= PAUSED_TIMEOUT_MS) {
121 return DcsState::PAUSED;
122 }
123
124 // ── AIRBORNE: Both wheels off ground (engine state irrelevant) ──
125 if (wowLeft == 0 && wowRight == 0) {
126 return DcsState::AIRBORNE;
127 }
128
129 // ── On ground: Determine hot/cold by engine RPM ──
130 // GROUND_HOT: Both engines >= 50% RPM
131 // GROUND_COLD: At least one engine < 50% RPM
132 if (rpmLeft >= RPM_THRESHOLD && rpmRight >= RPM_THRESHOLD) {
133 return DcsState::GROUND_HOT;
134 } else {
135 return DcsState::GROUND_COLD;
136 }
137}
138
143bool checkDcsRunning() {
144 return getDcsState() != DcsState::EXITED;
145}

154void onDcsUpdateCounterChange(unsigned int newValue) {

159// Aircraft name - empty when no mission or DCS in menu
160void onAcftNameChange(char* newValue) {
161 if (newValue != nullptr && newValue[0] != '\0') {
162 strncpy(acftName, newValue, sizeof(acftName) - 1);
163 acftName[sizeof(acftName) - 1] = '\0';
164 } else {
165 acftName[0] = '\0';
166 }
167}
