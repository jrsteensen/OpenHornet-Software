154void onHookBypassSwChange(unsigned int newValue) {
155 if (newValue == hookBypassState) {
156 return; //no state change, no change to magnet.
157 } else {
158 switch (newValue) {
159 case 0: // Switch is down in carrier position turn off mag.
160 digitalWrite(HOOK_FIELD, LOW);
161 hookBypassMagState = LOW;
162 break;
163 case 1: // Switch is up in field position, turn on mag.
164 digitalWrite(HOOK_FIELD, HIGH);
165 hookBypassMagState = HIGH;
166 break;
167 default:
168 break;
169 } hookBypassState = newValue;
170 }
171} DcsBios::IntegerBuffer hookBypassSwBuffer(0x7480, 0x4000, 14, onHookBypassSwChange);
172
177void onHookLeverChange(unsigned int newValue) {
178 hookLeverState = newValue;
179 hookLeverTime = millis();
180} DcsBios::IntegerBuffer hookLeverBuffer(0x74a0, 0x0200, 9, onHookLeverChange);

186void onLaunchBarSwChange(unsigned int newValue) {
187 if (newValue == launchBarState) {
188 return; //no state change, no launch bar magnet state update.
189 } else {
190 switch (newValue) {
191 case 0: // Launch bar switch turned off
192 digitalWrite(LBAR_RET, LOW);
193 launchBarMagState = LOW;
194 break;
195 case 1: // Launch bar switch turned on, only lock launch bar down if weight on wheels
196 if (wowLeft == wowRight == wowNose == true) {
197 digitalWrite(LBAR_RET, HIGH);
198 launchBarMagState = HIGH;
199 } else {
200 digitalWrite(LBAR_RET, LOW);
201 launchBarMagState = LOW;
202 }
203 break;
204 default:
205 break;
206 }
207 }
208 launchBarState = newValue;
209} DcsBios::IntegerBuffer launchBarSwBuffer(0x7480, 0x2000, 13, onLaunchBarSwChange);

260void loop() {
261
262 //Run DCS Bios loop function
263 DcsBios::loop();
264
273 if (launchBarMagState == HIGH) {
274 switch (launchBarState) {
275 case LOW: //launch bar switch in retract
276 //do nothing the state change method will retract the launch bar
277 break;
278 case 1: //launch bar switch in extend
279 if (wowLeft == wowRight == wowNose == false){ // no weight on wheels retract launch bar
280 digitalWrite(LBAR_RET, LOW);
281 launchBarMagState = LOW; // mag is off
282 break;
283 }
284 if ((LBAR_SW_AUTORETRACT == true) && (rpmL >= 85 && rpmR >= 85)) { //If Launch bar auto retract on throttle is true and both engines over 80% rpm turn off mag
285 digitalWrite(LBAR_RET, LOW);
286 launchBarMagState = LOW; // mag is off
287 }
288 break;
289 }
290 }
291
300 if ((hookBypassMagState == HIGH) && (wowLeft == wowRight == wowNose == false) && (hookLeverState == LOW)) { // If the hook bypass mag is on, with weight off wheels, and the hook lever is down.
301 if ((millis() - hookLeverTime ) > HOOK_DELAY) { // if longer than the hook override auto cancel delay
302 digitalWrite(HOOK_FIELD, LOW);
303 hookBypassMagState = LOW;
304 } else {
305 //wait for time to pass to ensure hook lever isn't raised before the auto-cancel time is met.
306 }
307 }
