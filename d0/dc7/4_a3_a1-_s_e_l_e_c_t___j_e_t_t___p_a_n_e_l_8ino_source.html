140void onHookBypassSwChange(unsigned int newValue) {
141 if (newValue == hookBypassState) {
142 return; //no state change, no change to magnet.
143 } else {
144 switch (newValue) {
145 case 0: // Switch is down in carrier position turn off mag.
146 digitalWrite(HOOK_FIELD, LOW);
147 break;
148 case 1: // Switch is up in field position, turn on mag.
149 digitalWrite(HOOK_FIELD, HIGH);
150 break;
151 default:
152 break;
153 } hookBypassState = newValue;
154 }
155} DcsBios::IntegerBuffer hookBypassSwBuffer(FA_18C_hornet_HOOK_BYPASS_SW, onHookBypassSwChange);
156
160void onLaunchBarSwChange(unsigned int newValue) {
161 if (newValue == launchBarState) {
162 return; //no state change, no launch bar magnet state update.
163 } else {
164 switch (newValue) {
165 case 0: // Launch bar switch turned off
166 digitalWrite(LBAR_RET, LOW);
167 break;
168 case 1: // Launch bar switch turned on, only lock launch bar down if weight on wheels
169 digitalWrite(LBAR_RET, HIGH);
170 break;
171 default:
172 break;
173 }
174 }
175 launchBarState = newValue;
176} DcsBios::IntegerBuffer launchBarSwBuffer(FA_18C_hornet_LAUNCH_BAR_SW, onLaunchBarSwChange);
