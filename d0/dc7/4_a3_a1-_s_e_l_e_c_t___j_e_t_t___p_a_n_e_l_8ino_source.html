138void onHookBypassSwChange(unsigned int newValue) {
139 if (newValue == hookBypassState) {
140 return; //no state change, no change to magnet.
141 } else {
142 switch (newValue) {
143 case 0: // Switch is down in carrier position turn off mag.
144 digitalWrite(HOOK_FIELD, LOW);
145 break;
146 case 1: // Switch is up in field position, turn on mag.
147 digitalWrite(HOOK_FIELD, HIGH);
148 break;
149 default:
150 break;
151 } hookBypassState = newValue;
152 }
153} DcsBios::IntegerBuffer hookBypassSwBuffer(FA_18C_hornet_HOOK_BYPASS_SW, onHookBypassSwChange);
154
158void onLaunchBarSwChange(unsigned int newValue) {
159 if (newValue == launchBarState) {
160 return; //no state change, no launch bar magnet state update.
161 } else {
162 switch (newValue) {
163 case 0: // Launch bar switch turned off
164 digitalWrite(LBAR_RET, LOW);
165 break;
166 case 1: // Launch bar switch turned on, only lock launch bar down if weight on wheels
167 digitalWrite(LBAR_RET, HIGH);
168 break;
169 default:
170 break;
171 }
172 }
173 launchBarState = newValue;
174} DcsBios::IntegerBuffer launchBarSwBuffer(FA_18C_hornet_LAUNCH_BAR_SW, onLaunchBarSwChange);
