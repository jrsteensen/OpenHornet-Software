259// Compute minimal signed difference of angles in degrees (wrap-around 0..360)
263static float wrapDiffDeg(float aDeg, float bDeg)
264{
265 float d = aDeg - bDeg;
266 while (d > 180.0f) d -= 360.0f;
267 while (d < -180.0f) d += 360.0f;
268 return d;
269}

281// Symmetric axis mapping with deadzone and post-deadzone renormalization
285static int16_t mapDegToHID_Symmetric(float deltaDeg, float halfRangeDeg, float deadzoneDeg)
286{
287 // Clamp
288 if (deltaDeg > halfRangeDeg) deltaDeg = halfRangeDeg;
289 if (deltaDeg < -halfRangeDeg) deltaDeg = -halfRangeDeg;
290
291 // Deadzone
292 float a = fabs(deltaDeg);
293 if (deadzoneDeg < 0) deadzoneDeg = 0;
294 if (deadzoneDeg > halfRangeDeg) deadzoneDeg = halfRangeDeg;
295
296 if (a <= deadzoneDeg) return 0;
297
298 // Renormalize to keep full output travel after deadzone
299 float scaled = (a - deadzoneDeg) / (halfRangeDeg - deadzoneDeg);
300 float signedNorm = (deltaDeg < 0 ? -scaled : scaled);
301
302 int32_t v = (int32_t)(signedNorm * 32767.0f);
303 if (v > HID_MAX) v = HID_MAX;
304 if (v < HID_MIN) v = HID_MIN;
305 return (int16_t)v;
306}

308// Asymmetric axis mapping with deadzone and post-deadzone renormalization
309// deltaDeg >= 0 uses rangePosDeg (push/forward), deltaDeg < 0 uses rangeNegDeg (pull/aft)
313static int16_t mapDegToHID_Asymmetric(float deltaDeg, float rangePosDeg, float rangeNegDeg, float deadzoneDeg)
314{
315 float range = (deltaDeg >= 0) ? rangePosDeg : rangeNegDeg;
316 if (range < 0.001f) return 0;
317
318 // Clamp using side-specific range
319 if (deltaDeg > rangePosDeg) deltaDeg = rangePosDeg;
320 if (deltaDeg < -rangeNegDeg) deltaDeg = -rangeNegDeg;
321
322 // Deadzone
323 float a = fabs(deltaDeg);
324 if (deadzoneDeg < 0) deadzoneDeg = 0;
325 if (deadzoneDeg > range) deadzoneDeg = range;
326
327 if (a <= deadzoneDeg) return 0;
328
329 // Renormalize to keep full output travel after deadzone
330 float scaled = (a - deadzoneDeg) / (range - deadzoneDeg);
331 float signedNorm = (deltaDeg < 0 ? -scaled : scaled);
332
333 int32_t v = (int32_t)(signedNorm * 32767.0f);
334 if (v > HID_MAX) v = HID_MAX;
335 if (v < HID_MIN) v = HID_MIN;
336 return (int16_t)v;
337}

340// EEPROM helpers
341// =====================================================
345static void eepromWriteU32(int addr, uint32_t v)
346{
347 EEPROM.update(addr + 0, (uint8_t)(v >> 0));
348 EEPROM.update(addr + 1, (uint8_t)(v >> 8));
349 EEPROM.update(addr + 2, (uint8_t)(v >> 16));
350 EEPROM.update(addr + 3, (uint8_t)(v >> 24));
351}
356static uint32_t eepromReadU32(int addr)
357{
358 uint32_t v = 0;
359 v |= (uint32_t)EEPROM.read(addr + 0) << 0;
360 v |= (uint32_t)EEPROM.read(addr + 1) << 8;
361 v |= (uint32_t)EEPROM.read(addr + 2) << 16;
362 v |= (uint32_t)EEPROM.read(addr + 3) << 24;
363 return v;
364}

366// Returns true if EEPROM contains a valid calibration
370static bool loadCentersFromEEPROM(uint32_t &r, uint32_t &p)
371{
372 uint32_t magic = eepromReadU32(EEPROM_ADDR_MAGIC);
373 if (magic != EEPROM_MAGIC) return false;
374
375 r = eepromReadU32(EEPROM_ADDR_ROLL);
376 p = eepromReadU32(EEPROM_ADDR_PITCH);
377
378 // Sanity check for 21-bit values
379 if ((r & 0x1FFFFF) != r) return false;
380 if ((p & 0x1FFFFF) != p) return false;
381
382 return true;
383}

420// =====================================================
421// MT6835 CRC (CRC-8 poly 0x07, MSB-first over 24 bits)
422// The CRC covers 24 bits: ANGLE[20:0] + STATUS[2:0]
423// =====================================================
427static uint8_t mt_crc8_24bits_msb(uint32_t data24)
428{
429 uint8_t crc = 0x00;
430 for (int i = 23; i >= 0; --i)
431 {
432 uint8_t bit = (data24 >> i) & 0x01;
433 uint8_t c7 = (crc >> 7) & 0x01;
434 crc <<= 1;
435 if (c7 ^ bit) crc ^= 0x07;
436 }
437 return crc;
438}

440// Build MT6835 24-bit frame: [cmd(4) | addr(12) | data(8)]
444static inline uint32_t mt_frame(uint8_t cmd4, uint16_t addr12, uint8_t data8)
445{
446 cmd4 &= 0x0F;
447 addr12 &= 0x0FFF;
448 return ((uint32_t)cmd4 << 20) | ((uint32_t)addr12 << 8) | (uint32_t)data8;
449}

451// Burst read angle: reads registers 0x003..0x006 in one CS low
452static MT6835_Result readMT6835_burstAngle(uint8_t csPin)
453{
454 MT6835_Result res{};
455 res.angle21 = 0;
456 res.status3 = 0;
457 res.crc_ok = true;
458
459 SPI.beginTransaction(spiMT);
460 digitalWrite(csPin, LOW);
461
462 // Send burst command starting at 0x003
463 uint32_t frame = mt_frame(MT_CMD_BURST_ANG, 0x003, 0x00);
464 uint8_t tx0 = (frame >> 16) & 0xFF;
465 uint8_t tx1 = (frame >> 8) & 0xFF;
466 uint8_t tx2 = (frame >> 0) & 0xFF;
467
468 (void)SPI.transfer(tx0);
469 (void)SPI.transfer(tx1);
470 uint8_t r003 = SPI.transfer(tx2);
471 uint8_t r004 = SPI.transfer(0x00);
472 uint8_t r005 = SPI.transfer(0x00);
473 uint8_t r006 = SPI.transfer(0x00);
474
475 digitalWrite(csPin, HIGH);
476 SPI.endTransaction();
477
478 // Reconstruct ANGLE[20:0] and STATUS[2:0]
479 uint32_t angle21 =
480 ((uint32_t)r003 << 13) |
481 ((uint32_t)r004 << 5) |
482 ((uint32_t)(r005 >> 3) & 0x1F);
483
484 uint8_t status3 = r005 & 0x07;
485
486 res.angle21 = angle21 & 0x1FFFFF;
487 res.status3 = status3;
488
489#if MT6835_USE_CRC
490 uint32_t data24 = ((res.angle21 & 0x1FFFFF) << 3) | (uint32_t)(res.status3 & 0x07);
491 uint8_t crc = mt_crc8_24bits_msb(data24);
492 res.crc_ok = (crc == r006);
493#else
494 (void)r006;
495 res.crc_ok = true;
496#endif
497
498 return res;
499}

501// =====================================================
502// Grip read (24-bit) - Mode 3
503// =====================================================
507static uint32_t readGrip24()
508{
509 SPI.beginTransaction(spiGRIP);
510 digitalWrite(PIN_CS_GRIP, LOW);
511 delayMicroseconds(1);
512
513 uint8_t b2 = SPI.transfer(0x00);
514 uint8_t b1 = SPI.transfer(0x00);
515 uint8_t b0 = SPI.transfer(0x00);
516
517 digitalWrite(PIN_CS_GRIP, HIGH);
518 SPI.endTransaction();
519
520 return ((uint32_t)b2 << 16) | ((uint32_t)b1 << 8) | (uint32_t)b0;
521}

523// =====================================================
524// POV hat from Trim bits
525// Note: Up/Down were inverted, so we swap those here.
526// =====================================================
530static int16_t computeHatFromTrim(uint32_t grip24)
531{
532 bool up = getBit24(grip24, BIT_TRIM_NOSE_DOWN);
533 bool down = getBit24(grip24, BIT_TRIM_NOSE_UP);
534 bool left = getBit24(grip24, BIT_TRIM_LWD);
535 bool right = getBit24(grip24, BIT_TRIM_RWD);
536
537 // Joystick library: -1 = released, otherwise 0..315 in 45° steps
538 if (!up && !down && !left && !right) return -1;
539
540 if (up && right) return 45;
541 if (up && left) return 315;
542 if (down && right) return 135;
543 if (down && left) return 225;
544
545 if (up) return 0;
546 if (right) return 90;
547 if (down) return 180;
548 if (left) return 270;
549
550 return -1;
551}

636// =====================================================
637// Setup
638// =====================================================
642void setup()
643{
644#if DEBUG_SERIAL
645 Serial.begin(115200);
646 delay(400);
647#endif
648
649 // CS pins idle high
650 pinMode(PIN_CS_ROLL, OUTPUT);
651 pinMode(PIN_CS_PITCH, OUTPUT);
652 pinMode(PIN_CS_GRIP, OUTPUT);
653
654 digitalWrite(PIN_CS_ROLL, HIGH);
655 digitalWrite(PIN_CS_PITCH, HIGH);
656 digitalWrite(PIN_CS_GRIP, HIGH);
657
658 SPI.begin();
659
660 // HID init
661 Joystick.setXAxisRange(HID_MIN, HID_MAX);
662 Joystick.setYAxisRange(HID_MIN, HID_MAX);
663 Joystick.begin();
664
665 // Load absolute centers from EEPROM if valid; otherwise initialize once from current position
666 centersValid = loadCentersFromEEPROM(rollCenter, pitchCenter);
667
668#if DEBUG_SERIAL
669 if (centersValid) Serial.println(F("EEPROM centers loaded (valid)."));
670 else Serial.println(F("EEPROM centers not valid (will init from boot position)."));
671#endif
672
673 if (!centersValid)
674 {
675 delay(50);
676 for (int i = 0; i < 8; i++)
677 {
678 MT6835_Result r = readMT6835_burstAngle(PIN_CS_ROLL);
679 if (r.crc_ok) rollCenter = r.angle21;
680
681 MT6835_Result p = readMT6835_burstAngle(PIN_CS_PITCH);
682 if (p.crc_ok) pitchCenter = p.angle21;
683
684 delay(2);
685 }
686
687 saveCentersToEEPROM(rollCenter, pitchCenter);
688 centersValid = true;
689
690#if DEBUG_SERIAL
691 Serial.println(F("EEPROM centers initialized from boot position (now valid)."));
692#endif
693 }
694
695#if DEBUG_SERIAL
696 debugDumpEEPROMCenters();
697#endif
698}

701// Main loop
702// =====================================================
706void loop()
707{
708 // ----- Read MT6835 sensors -----
709 MT6835_Result roll = readMT6835_burstAngle(PIN_CS_ROLL);
710 MT6835_Result pitch = readMT6835_burstAngle(PIN_CS_PITCH);
711
712 // Keep last valid degrees if CRC fails (avoid jumps)
713 static float rollDegPrev = 0.0f;
714 static float pitchDegPrev = 0.0f;
715
716 float rollDeg = rollDegPrev;
717 float pitchDeg = pitchDegPrev;
718
719 if (roll.crc_ok) { rollDeg = angle21ToDeg(roll.angle21); rollDegPrev = rollDeg; }
720 if (pitch.crc_ok) { pitchDeg = angle21ToDeg(pitch.angle21); pitchDegPrev = pitchDeg; }
721
722 // Compute deltas versus current centers
723 float rollCenterDeg = angle21ToDeg(rollCenter);
724 float pitchCenterDeg = angle21ToDeg(pitchCenter);
725
726 float rollDelta = wrapDiffDeg(rollDeg, rollCenterDeg);
727 float pitchDelta = wrapDiffDeg(pitchDeg, pitchCenterDeg);
728
729 if (INVERT_ROLL) rollDelta = -rollDelta;
730 if (INVERT_PITCH) pitchDelta = -pitchDelta;
731
732 // ----- Read grip (24-bit) -----
733 uint32_t grip24 = readGrip24();
734
735 // Your grip is active-low: 0 = pressed, 1 = released
736 // Invert so that 1 = pressed for our mapping.
737 grip24 = ~grip24 & 0xFFFFFF;
738
739 // ----- Recenter (SAFE 2s / FORCED 6s with correct behavior) -----
740 bool combo = recenterComboPressed(grip24);
741
742 if (combo)
743 {
744 if (!recenterArmed)
745 {
746 recenterArmed = true;
747 recenterStartMs = millis();
748 recenterDoneThisHold = false;
749 recenterSafeTriedThisHold = false;
750 }
751 else if (!recenterDoneThisHold)
752 {
753 uint32_t held = millis() - recenterStartMs;
754
755 // Forced recenter has priority once you reach the forced hold time
756 if (held >= RECENTER_FORCE_HOLD_MS)
757 {
758 if (roll.crc_ok && pitch.crc_ok)
759 {
760 rollCenter = roll.angle21;
761 pitchCenter = pitch.angle21;
762 saveCentersToEEPROM(rollCenter, pitchCenter);
763 centersValid = true;
764
765#if DEBUG_SERIAL
766 Serial.println(F("RECENTER FORCED (saved)."));
767 debugDumpEEPROMCenters();
768#endif
769 }
770 recenterDoneThisHold = true;
771 }
772 // Try safe recenter once at 2s; if refused, keep holding to allow forced at 6s
773 else if (held >= RECENTER_HOLD_MS && !recenterSafeTriedThisHold)
774 {
775 recenterSafeTriedThisHold = true;
776
777 // If EEPROM calibration was not valid, treat SAFE as forced init
778 if (!centersValid)
779 {
780 if (roll.crc_ok && pitch.crc_ok)
781 {
782 rollCenter = roll.angle21;
783 pitchCenter = pitch.angle21;
784 saveCentersToEEPROM(rollCenter, pitchCenter);
785 centersValid = true;
786
787#if DEBUG_SERIAL
788 Serial.println(F("RECENTER (EEPROM invalid -> forced init) saved."));
789 debugDumpEEPROMCenters();
790#endif
791 }
792 recenterDoneThisHold = true;
793 }
794 else
795 {
796 // Safe recenter: allow only if near current center
797 float rErr = fabs(rollDelta);
798 float pErr = fabs(pitchDelta);
799 bool okPos = (rErr <= RECENTER_MAX_ROLL_ERR_DEG) && (pErr <= RECENTER_MAX_PITCH_ERR_DEG);
800
801 if (okPos && roll.crc_ok && pitch.crc_ok)
802 {
803 rollCenter = roll.angle21;
804 pitchCenter = pitch.angle21;
805 saveCentersToEEPROM(rollCenter, pitchCenter);
806
807#if DEBUG_SERIAL
808 Serial.print(F("RECENTER SAFE OK (saved). rErr="));
809 Serial.print(rErr, 3);
810 Serial.print(F(" pErr="));
811 Serial.println(pErr, 3);
812 debugDumpEEPROMCenters();
813#endif
814 recenterDoneThisHold = true;
815 }
816 else
817 {
818#if DEBUG_SERIAL
819 Serial.print(F("RECENTER SAFE REFUSED. rErr="));
820 Serial.print(rErr, 3);
821 Serial.print(F(" pErr="));
822 Serial.print(pErr, 3);
823 Serial.println(F(" -> Keep holding to 6s for FORCED recenter."));
824#endif
825 // Do not set recenterDoneThisHold here: allow forced recenter later
826 }
827 }
828 }
829 }
830 }
831 else
832 {
833 // Combo released: reset hold state
834 recenterArmed = false;
835 recenterDoneThisHold = false;
836 recenterSafeTriedThisHold = false;
837 }
838
839 // ----- Map axes to HID -----
840 // Roll: apply sensor->mechanical scaling, then symmetric mapping + deadzone
841 float rollDeltaMech = rollDelta * ROLL_SENSOR_TO_MECH_SCALE;
842 int16_t x = mapDegToHID_Symmetric(rollDeltaMech, ROLL_HALF_DEG, DEADZONE_ROLL_DEG);
843
844 // Pitch: asymmetric mapping + deadzone
845 // Convention here: pitchDelta > 0 => forward (push / "piquer"), pitchDelta < 0 => aft (pull / "cabrer")
846 int16_t y = mapDegToHID_Asymmetric(
847 pitchDelta,
848 /*rangePosDeg*/ PITCH_PIQUER_DEG,
849 /*rangeNegDeg*/ PITCH_CABRER_DEG,
850 DEADZONE_PITCH_DEG
851 );
852
853 // ----- Hat + Buttons -----
854 int16_t hat = computeHatFromTrim(grip24);
855
856 // Update HID only when changed (reduces USB traffic)
857 if (x != lastX) { Joystick.setXAxis(x); lastX = x; }
858 if (y != lastY) { Joystick.setYAxis(y); lastY = y; }
859
860 if (grip24 != lastGrip24)
861 {
862 updateButtonsFromGrip(grip24);
863 lastGrip24 = grip24;
864 }
865
866 if (hat != lastHat)
867 {
868 Joystick.setHatSwitch(0, hat);
869 lastHat = hat;
870 }
871
872 // ----- Debug -----
873 debugPrint(grip24, roll, rollDeg, rollDelta, pitch, pitchDeg, pitchDelta);
874
875 delay(1);
876}
