113void onFuelDumpSwChange(unsigned int newValue) {
114 if (newValue == fuelDumpState) {
115 return; // no state change, no magnet update.
116 } else {
117 switch (newValue) {
118 case 0: // switch turned off manually in game...disengage magnet.
119 digitalWrite(DUMP_MAG, LOW);
120 break;
121 case 1:
122 digitalWrite(DUMP_MAG, HIGH); // switch turned on, either physically or virtually, engage magnet. May get overridden by bingo logic in loop.
123 break;
124 default:
125 break;
126 }
127 fuelDumpState = newValue;
128 }
129} DcsBios::IntegerBuffer fuelDumpSwBuffer(FA_18C_hornet_FUEL_DUMP_SW, onFuelDumpSwChange);
