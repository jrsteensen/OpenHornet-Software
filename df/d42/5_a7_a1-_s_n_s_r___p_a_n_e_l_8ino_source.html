136void onLtdRSwChange(unsigned int newValue) {
137 switch (newValue) {
138 case 0:
139 digitalWrite(LTDR_ARM_MAG, LOW); // switch turned off physically or virtually in the sim, turn off mag.
140 break;
141 case 1:
142 digitalWrite(LTDR_ARM_MAG, HIGH); // hold the mag-swtich in the 'ARM' position.
143 }
144}
145DcsBios::IntegerBuffer ltdRSwBuffer(FA_18C_hornet_LTD_R_SW, onLtdRSwChange);
