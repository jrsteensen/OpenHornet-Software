159void onMasterModeAgLtChange(unsigned int newValue) {
160 airToGroundLight = newValue;
161 if (airToGroundLight == 0 && ltdrArmMagEngaged == true) { // light off and the LTD/R mag-switch is on.
162 digitalWrite(LTDR_ARM_MAG, LOW); // turn off the mag-switch.
163 ltdrArmMagEngaged = false; // remember that the switch is released.
164 }
165}
166DcsBios::IntegerBuffer masterModeAgLtBuffer(0x740c, 0x0400, 10, onMasterModeAgLtChange);
167
169void onGearLeverChange(unsigned int newValue) {
170 if (newValue == 0 && ltdrArmMagEngaged == true) { // landing gear lever lowered and mag-switch is on.
171 digitalWrite(LTDR_ARM_MAG, LOW); // Landing gear handle lowered, turn off LTD/R Mag.
172 ltdrArmMagEngaged = false; // remember that the switch is released.
173 }
174}
175DcsBios::IntegerBuffer gearLeverBuffer(0x747e, 0x1000, 12, onGearLeverChange);
176
178void onLtdRSwChange(unsigned int newValue) {
179 switch (newValue) {
180 case 0:
181 digitalWrite(LTDR_ARM_MAG, LOW); // switch turned off physically or virtually in the sim, turn off mag.
182 ltdrArmMagEngaged = false; // remember that the switch is released.
183 break;
184 case 1:
185 if ((leftGearDown == rightGearDown == noseGearDown == false) && (airToGroundLight == true)) { // if the gear is up and A/G light On the mag-switch will hold.
186 digitalWrite(LTDR_ARM_MAG, HIGH); // hold the mag-swtich in the 'ARM' position.
187 ltdrArmMagEngaged = true; // remember that the mag-switch is holding.
188 break;
189 }
190 }
191}
192DcsBios::IntegerBuffer ltdRSwBuffer(0x74c8, 0x4000, 14, onLtdRSwChange);
193
199void onFlirSwChange(unsigned int newValue) {
200 switch (newValue) {
201 case 0:
202 if (ltdrArmMagEngaged == true) {
203 digitalWrite(LTDR_ARM_MAG, LOW); // flir switch turned off, turn off LTD/R Mag.
204 ltdrArmMagEngaged = false; // remember that the switch is released.
205 }
206 break;
207 case 1:
208 break;
209 case 2:
210 break;
211 }
212}
213DcsBios::IntegerBuffer flirSwBuffer(0x74c8, 0x3000, 12, onFlirSwChange);
