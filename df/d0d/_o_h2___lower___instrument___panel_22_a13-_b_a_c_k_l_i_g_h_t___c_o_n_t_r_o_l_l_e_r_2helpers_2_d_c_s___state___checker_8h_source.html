60// ─────────────────────────────────────────────────────────────────────────────
61// Note: This is a lightweight variant of DCS_State_Checker for use in the
62// 2A13-BACKLIGHT_CONTROLLER. It uses heartbeat-only detection to avoid the
63// StringBuffer and extra IntegerBuffer overhead of the full version, which
64// would exceed the Arduino Mega 2560's SRAM budget for this sketch.
65// RPM/WoW/aircraft-name states are not needed here; Board.h only requires
66// the EXITED / PAUSED / active distinction.
67// ─────────────────────────────────────────────────────────────────────────────
68
69enum class DcsState {
70 EXITED, // DCS not running, in menu, or crashed (no heartbeat for 30 min)
71 PAUSED, // Mission loaded but paused (no heartbeat for 10 s)
72 GROUND_COLD, // Heartbeat active - DCS is running (used as generic "active" state)
73 GROUND_HOT, // (not used in this variant)
74 AIRBORNE // (not used in this variant)
75};
76
77// ─────────────────────────────────────────────────────────────────────────────
78// Heartbeat tracking
79// ─────────────────────────────────────────────────────────────────────────────
80static unsigned int currDcsHeartbeat = 0;
81static unsigned int prevDcsHeartbeat = 0;
82static unsigned long dcsLastUpdateTime = 0;
83
84static const unsigned long PAUSED_TIMEOUT_MS = 10000; // 10 seconds -> PAUSED
85static const unsigned long EXITED_TIMEOUT_MS = 1800000; // 30 minutes -> EXITED
86
87// ─────────────────────────────────────────────────────────────────────────────
88// State determination function
89// ─────────────────────────────────────────────────────────────────────────────
90
98DcsState getDcsState() {
99 unsigned long currentTime = millis();
100 unsigned long timeSinceUpdate = currentTime - dcsLastUpdateTime;
101
102 if (currDcsHeartbeat != prevDcsHeartbeat) {
103 dcsLastUpdateTime = currentTime;
104 prevDcsHeartbeat = currDcsHeartbeat;
105 timeSinceUpdate = 0;
106 }
107
108 if (timeSinceUpdate >= EXITED_TIMEOUT_MS) return DcsState::EXITED;
109 if (timeSinceUpdate >= PAUSED_TIMEOUT_MS) return DcsState::PAUSED;
110 return DcsState::AIRBORNE;
111}
112
117bool checkDcsRunning() {
118 return getDcsState() != DcsState::EXITED;
119}

122// DCS-BIOS Callback — heartbeat counter only

124void onDcsUpdateCounterChange(unsigned int newValue) {
125 currDcsHeartbeat = newValue;
