56class SwitchMultiPosDebounce {
57private:
58 const char* msg_;
59 const byte* pins_;
60 char numberOfPins_;
61 char lastState_;
62 bool reverse_;
63 char debounceSteadyState_;
64 unsigned long debounceDelay_;
65 unsigned long lastDebounceTime = 0;
66
71 char readState() {
72 unsigned char ncPinIdx = lastState_;
73 for (unsigned char i = 0; i < numberOfPins_; i++) {
74 if (pins_[i] == DcsBios::PIN_NC)
75 ncPinIdx = i;
76 else {
77 if (digitalRead(pins_[i]) == LOW && reverse_ == false) return i;
78 else if (digitalRead(pins_[i]) == HIGH && reverse_ == true) return i;
79 }
80 }
81 return ncPinIdx;
82 }
83
87 void resetState() {
88 lastState_ = (lastState_ == 0) ? -1 : 0;
89 }
90
94 void pollInput() {
95 char state = readState();
96
97 unsigned long now = millis();
98 if (state != debounceSteadyState_) {
99 lastDebounceTime = now;
100 debounceSteadyState_ = state;
101 }
102 if ((now - lastDebounceTime) >= debounceDelay_) {
103 if (state != lastState_) {
104 char buf[7];
105 utoa(state, buf, 10);
106 if (DcsBios::tryToSendDcsBiosMessage(msg_, buf))
107 lastState_ = state;
108 }
109 }
110 }
111public:
120 SwitchMultiPosDebounce(const char* msg, const byte* pins, char numberOfPins, bool reverse = false, unsigned long debounceDelay = 50) {
121 msg_ = msg;
122 pins_ = pins;
123 reverse_ = reverse;
124 numberOfPins_ = numberOfPins;
125 unsigned char i;
126 for (i = 0; i < numberOfPins; i++) {
127 if (pins[i] != DcsBios::PIN_NC)
128 pinMode(pins[i], INPUT_PULLUP);
129 }
130 lastState_ = readState();
131 debounceSteadyState_ = lastState_;
132 debounceDelay_ = debounceDelay;
133 }
