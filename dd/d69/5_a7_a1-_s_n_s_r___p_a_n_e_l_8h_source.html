71 char readState() {
72 unsigned char ncPinIdx = lastState_;
73 for (unsigned char i = 0; i < numberOfPins_; i++) {
74 if (pins_[i] == DcsBios::PIN_NC)
75 ncPinIdx = i;

164class SwitchRadar {
165private:
166 const unsigned int pullDelayMs = 200;
167 const char* msg_;
168 const char* pullMsg_;
169 char pullPositionNumber_;
170 const byte* pins_;
171 char numberOfPins_;
172 char lastState_;
173 bool reverse_;
174 char debounceSteadyState_;
175 unsigned long debounceDelay_;
176 unsigned long lastDebounceTime = 0;
177
181 enum switchPullStateEnum {
182 stOFF_PUSHED = 0,
183 stOFF_PULLED = 1,
184 stON_PULLED = 2
185 };
186
187 switchPullStateEnum switchPullState_;
188 unsigned long lastSwitchStateTime;
189
194 char readState() {
195 unsigned char ncPinIdx = lastState_;
196 for (unsigned char i = 0; i < numberOfPins_; i++) {
197 if (pins_[i] == DcsBios::PIN_NC)
198 ncPinIdx = i;
199 else {
200 if (digitalRead(pins_[i]) == LOW && reverse_ == false) return i;
201 else if (digitalRead(pins_[i]) == HIGH && reverse_ == true) return i;
202 }
203 }
204 return ncPinIdx;
205 }
206
210 void resetState() {
211 lastState_ = (lastState_ == 0) ? -1 : 0;
212 }
213
217 void pollInput() {
218 char state = readState();
219
220 unsigned long now = millis();
221 if (state != debounceSteadyState_) {
222 lastDebounceTime = now;
223 debounceSteadyState_ = state;
224 }
225 if ((now - lastDebounceTime) >= debounceDelay_) {
226 if (now - lastSwitchStateTime > pullDelayMs) {
227 if (state != lastState_) {
228 char buf[7];
229 utoa(state, buf, 10);
230 if (state != pullPositionNumber_) {
231 switch (switchPullState_) {
232 case stON_PULLED:
233 if (DcsBios::tryToSendDcsBiosMessage(pullMsg_, "1")) { //
234 switchPullState_ = stOFF_PULLED;
235 }
236 break;
237
238 case stOFF_PULLED:
239 if (DcsBios::tryToSendDcsBiosMessage(msg_, buf)) {
240 switchPullState_ = stOFF_PUSHED;
241 lastSwitchStateTime = millis();
242 lastState_ = state;
243 }
244 break;
245
246 case stOFF_PUSHED:
247 if (DcsBios::tryToSendDcsBiosMessage(msg_, buf)) {
248 lastState_ = state;
249 }
250 break;
251 }
252 } else { // state is in the locked out position, need to handle pull and rotate.
253 switch (switchPullState_) {
254 case stOFF_PUSHED:
255 if (DcsBios::tryToSendDcsBiosMessage(pullMsg_, "1")) {
256 switchPullState_ = stOFF_PULLED;
257 lastSwitchStateTime = millis(); // switch animation delay needed.
258 }
259 break;
260
261 case stOFF_PULLED:
262 if (DcsBios::tryToSendDcsBiosMessage(msg_, buf)) {
263 switchPullState_ = stON_PULLED;
264 lastState_ = state;
265 }
266 case stON_PULLED:
267 // no changes needed.
268 break;
269 }
270 }
271 }
272 }
273 }
274 }
